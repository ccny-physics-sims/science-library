Class: FBD

FBD

new FBD(position, howManyForces, showResultant, options)

Makes a free-body diagram.
Parameters:
Name Type Description
position p5.Vector The position vector.
howManyForces number Choose how many forces to display.
showResultant bool Display the resulting net force.
options object Pass in an object containing custom properties. (optional)
Properties:
Name Type Description
shape string Choose the shape of the base. ('rect' or 'circle')
shapeSize p5.Vector Customize the dimensions of the base shape.
mag array Array of numbers specifying the magnitude of each of the forces.
direction array Array of numbers specifying the direction of each of the forces.
labels array Array of strings specifying the labels for each of the forces.
offsets array Array of numbers specifying the offset value for each of the forces arrows in pixels.
FBD.display() method Draws the FBD object.
FBD.update() method Updates the FBD object.

Examples
// creates a box FBD
function setup() {
	canvas = createCanvas(500, 500);
	var bodyLocation = createVector(width / 3, height / 2);
	box_fbd = new FBD(bodyLocation, 3, true);
	box_fbd.mag = [175, 175, 75];
	box_fbd.direction = [0, -PI / 2, 0];
}

function draw() {
	box_fbd.update();
	box_fbd.display();
}
// creates a ball FBD
function setup() {
	canvas = createCanvas(500, 500);
	var bodyLocation = createVector(width / 3, height / 4);
  var options = { shape: 'circle' };
	ball_fbd = new FBD(bodyLocation, 2, true, options);
	ball_fbd.mag = [175, 175];
	ball_fbd.direction = [0, PI / 2];
}

function draw() {
	ball_fbd.update();
	ball_fbd.display();
}
// creates a ball FBD with some custom sizing and labels
function setup() {
	canvas = createCanvas(500, 500);
	var bodyLocation = createVector(width / 3, height / 4);
  var options = { shape: 'circle', 
		shapeSize: createVector(30,30),
        labels: [ 'Wind', 'Gravity'], 
		mag: [175, 175],
		direction: [0, PI / 2]		
       };
	ball_fbd = new FBD(bodyLocation, 2, true, options);
}

function draw() {
	ball_fbd.update();
	ball_fbd.display();
}
